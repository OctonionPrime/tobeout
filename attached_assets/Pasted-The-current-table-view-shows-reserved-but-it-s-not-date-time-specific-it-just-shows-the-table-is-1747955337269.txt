javascript// Instead of general "reserved" status
const getTableStatus = async (tableId) => {
  return "reserved"; // Too vague!
}

// Need time-specific status
const getTableStatusForDateTime = async (tableId, date, time) => {
  const reservation = await getReservation({
    tableId,
    date,
    time // Check if this specific time slot is booked
  });
  
  if (reservation) {
    return {
      status: 'reserved',
      guestName: reservation.guestName,
      guestCount: reservation.guests,
      timeSlot: `${reservation.time} - ${getEndTime(reservation)}`
    };
  }
  
  return { status: 'available' };
}
